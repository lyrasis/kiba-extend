def initialize(key, klass, for_job)
  @key = key
  @klass = klass
  @for_job = for_job
  @msg = ":#{key} cannot be used as a lookup in :#{for_job} because "\
    "its src_class (#{klass}) does not include "\
    "Kiba::Extend::Soures::Lookupable"
  super(msg)
end
