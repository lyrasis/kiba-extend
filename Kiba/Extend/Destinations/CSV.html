def fields
  return [] unless File.exist?(filename)

  csv ||= ::CSV.open(filename, "r", **csv_options)
  hdrs = csv.shift.headers
  close
  hdrs
end

def write(row)
  @csv ||= ::CSV.open(filename, "wb", **csv_options)
  @headers ||= row.keys
  verify_initial_headers
  order_headers
  @headers_written ||= begin
    csv << headers
    true
  end
  csv << row.fetch_values(*@headers)
end
