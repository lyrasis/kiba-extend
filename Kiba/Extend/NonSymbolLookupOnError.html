def initialize(filekey, for_job)
  @filekey = filekey
  @for_job = for_job
  @msg = "The `lookup_on` value in the registry entry hash for "\
    ":#{filekey} is not a Ruby Symbol. "\
    "Prepend a : to the field name to fix."
  super(msg)
end
