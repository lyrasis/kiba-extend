def output?(jobkey)
  begin
    reg = Kiba::Extend.registry.resolve(jobkey)
  rescue Dry::Container::KeyError
    return false
  end
  return true if File.exist?(reg.path)

  res = Kiba::Extend::Command::Run.job(jobkey)

  !(res.outrows == 0)
end
