def self.call
  Kiba::Extend::Fcar.chute
    .map do |mod, comment|
      formatted_comment = if comment.empty?
        nil
      else
        "  #{comment}"
      end
      [mod, formatted_comment].compact
        .join("\n")
    end.join("\n")
end
