def call
  headers = %i[table field]
  CSV.open(output, "w", headers: headers,
    write_headers: true) do |csv|
    entries.map { |entry| id_empty_fields(entry) }
      .flatten
      .each { |row| csv << row.values_at(*headers) }
  end

  puts "Wrote empty field report to #{output}"
end
