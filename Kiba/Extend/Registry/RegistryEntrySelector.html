def tagged_all(*args)
  tags = args.flatten.map(&:to_sym)
  tags.inject(Kiba::Extend.registry.entries) do |arr, tag|
    arr.select { |entry| entry.tags.any?(tag) }
  end
end

def tagged_any(*args)
  tags = args.flatten.map(&:to_sym)
  results = tags.each_with_object([]) do |arg, arr|
    arr << tagged(arg)
  end
  results.flatten.uniq
end
