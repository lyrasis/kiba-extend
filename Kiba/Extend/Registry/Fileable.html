def default_args(path = nil)
  if requires_path? && path.nil?
    fail Kiba::Extend::PathRequiredError.new(self)
  elsif path.nil?
    labeled_options
  else
    {path_key => path}.merge(labeled_options)
  end
end

def labeled_options
  if options_key && default_file_options
    {options_key => default_file_options}
  else
    {}
  end
end
