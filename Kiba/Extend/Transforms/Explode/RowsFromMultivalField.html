Overview

def process(row)
  other_fields = row.keys.reject { |k| k == @field }
  fieldval = row.fetch(@field, nil)
  fieldval = fieldval.nil? ? [] : fieldval.split(@delim)
  if fieldval.size > 1
    fieldval.each do |val|
      rowcopy = row.clone
      other_fields.each { |f| rowcopy[f] = rowcopy.fetch(f, nil) }
      rowcopy[@field] = val
      yield(rowcopy)
    end
    nil
  else
    row
  end
end
