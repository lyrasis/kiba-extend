Overview

Delete a field value if the arbitrary Lambda passed in evaluates to true

Examples:

Without delim

xform = Delete::FieldValueConditional.new(
  fields: %i[a b],
  lambda: ->(val, row) { row[:c] == val }
)

input = [
  {a: nil, b: "", c: "c"},
  {a: "c", b: "b", c: "c"},
  {a: "a|c", b: "c", c: "c"}
]
result = Kiba::StreamingRunner.transform_stream(input, xform)
  .map{ |row| row }
expected = [
  {a: nil, b: nil, c: "c"},
  {a: nil, b: "b", c: "c"},
  {a: "a|c", b: nil, c: "c"}
]
expect(result).to eq(expected)

With delim

xform = Delete::FieldValueConditional.new(
  fields: %i[a b],
  lambda: ->(val, row) { row[:c] == val },
  delim: "|"
)

input = [
  {a: nil, b: "", c: "c"},
  {a: "c", b: "b", c: "c"},
  {a: "a|c", b: "c", c: "c"}
]
result = Kiba::StreamingRunner.transform_stream(input, xform)
  .map{ |row| row }
expected = [
  {a: nil, b: nil, c: "c"},
  {a: nil, b: "b", c: "c"},
  {a: "a", b: nil, c: "c"}
]
expect(result).to eq(expected)
