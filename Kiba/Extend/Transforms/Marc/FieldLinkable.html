def select_fields(record, tags)
  all = candidate_fields(record, tags)
    .map { |field| add_linkage_data(field) }
  preferred(all)
    .map { |fldhsh| update_tag(fldhsh) }
    .map { |fldhsh| fldhsh[:datafield] }
end
